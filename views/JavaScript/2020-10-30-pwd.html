判断密码强度

// 判断输入密码的类型
CharMode(iN){
    // 数字
    if(iN >= 48 && iN <= 57)
        return 1;
    // 大写
    if(iN >= 65 && iN <= 90)
        return 2;
    // 小写
    if(iN >= 97 && iN <= 122)
        return 4;
    else return 8;
};

// 计算密码模式
bitTotal(num){
    let modes = 0;
    for(let i=0; i<4; i++){
        // 位与运算
        if(num & 1) modes++;
        num >>>= 1;  // 移位运算
    };
    return modes;

    /*
        移位运算
        >>> 向右移位
        例：
        num = 8; 8的二进制位1000
        num >>> 1; 向右移1位，为0100 就是4
    */
};

// 返回强度级别
checkStrong(sPW){
    if(sPW.length < 6) return 0;   // 密码太短, 不检测级别
    let Modes = 0;
    for(let i=0; i < sPW.length; i++){
        // 密码模式
        Modes |= this.CharMode(sPW.charCodeAt(i));
    };
    return this.bitTotal(Modes);
}

// 密码强度颜色
pwStrength(pwd){
    let Default_color = '#eeeeee';  // 默认颜色
    let Lcolor = '#FF0000'; // 低强度的颜色, 且只显示在最左边的单元格中
    let Mcolor = '#FF9900'; // 中等强度的颜色, 且只显示在左边的两个单元格中
    let Hcolor = '#33CC00'; // 高强度的颜色, 三个单元格都显示

    if(pwd == null || pwd == ''){
        Lcolor = Mcolor = Hcolor = Default_color;
    }else{
        var S_level = this.checkStrong(pwd);
        if(S_level >= 3){
            this.level_S = true;
        }
        switch (S_level){
            case 0:
                Lcolor = Mcolor = Hcolor = Default_color;
                break;
            case 1:
                Mcolor = Hcolor = Default_color;
                break;
            case 2:
                Lcolor = Mcolor;
                Hcolor = Default_color;
                break;
            default:
                Lcolor = Mcolor = Hcolor;
        }
    }
    document.getElementById('strength_L').style.background = Lcolor;
    document.getElementById('strength_M').style.background = Lcolor;
    document.getElementById('strength_H').style.background = Lcolor;
    return;
}
